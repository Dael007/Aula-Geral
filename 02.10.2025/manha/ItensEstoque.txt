create database DB_ALMOXARIFADO;

use DB_ALMOXARIFADO;

create table Localizacao(
ID int auto_increment primary key,
Nome varchar (50) not null,
Descricao text
);

create table CategoriaItens(
ID int auto_increment primary key,
Localizacao int not null,
Nome varchar (50) not null,
Descricao text,
FOREIGN key (Localizacao) references Localizacao(ID)
);

create table ItensEstoque(
ID INT auto_increment primary key,
nome varchar(100),
Categoria INT not null,
Descricao text,
Quantidade INT not null default 0,
Codigo varchar (30) not null,
NomeImagem varchar (60),
FOREIGN key (Categoria) references CategoriaItens(ID)
);
